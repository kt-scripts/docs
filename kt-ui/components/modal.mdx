```lua
local ok = exports.kt_ui:openModal({
  title = 'Sell vehicle?',
  body  = 'You will receive $5,000. This cannot be undone.',
  confirmLabel = 'Sell',
  cancelLabel  = 'Cancel',
})

if ok then print('confirmed') end
```
